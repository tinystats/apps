my_covariance <- function(x,y){
  m_x <- mean(x)
  m_y <- mean(y)

  deviation_x <- x - m_x
  deviation_y <- y - m_y
  
  crossproduct <- deviation_x * deviation_y

  covariance <-  sum(crossproduct) / (length(x)-1) # Divides by N-1
  return(covariance)
}

my_correlation <- function(x,y){
 correlation <- my_covariance(x,y) / (sd(x)*sd(y))
 return(correlation)
}
